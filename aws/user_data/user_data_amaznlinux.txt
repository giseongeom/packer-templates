#!/usr/bin/env bash

# workaround for 'Sorry, You Must Have a Tty to Run Sudo' problem
# See https://www.shell-tips.com/2014/09/08/sudo-sorry-you-must-have-a-tty-to-run-sudo/
sed -i -e 's/^Defaults.*requiretty/# Defaults requiretty/g' /etc/sudoers

# disable reverse DNS lookup
# See https://injustfiveminutes.com/2013/03/13/fixing-ssh-login-long-delay/
sed -i -e 's/^#UseDNS.*yes/UseDNS no/g' /etc/ssh/sshd_config

sleep 120
